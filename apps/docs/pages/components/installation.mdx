# Installation

<Steps>
### Install dependencies
<Tabs items={["pnpm", "npm", "yarn"]}>
  <Tabs.Tab>

    ```sh
    pnpm i @crossed/ui @crossed/next-adapter
    pnpm i --save-dev modify-selectors
    ```

  </Tabs.Tab>
  <Tabs.Tab>

    ```sh
    npm i @crossed/ui @crossed/next-adapter
    npm i --save-dev modify-selectors
    ```

  </Tabs.Tab>
  <Tabs.Tab>
  
    ```sh
    yarn add @crossed/ui @crossed/next-adapter
    yarn add --dev modify-selectors
    ```
  
  </Tabs.Tab>
</Tabs>

### Tailwindcss
Add in tailwind config content to parse

```javascript
import type { Config } from 'tailwindcss';

const config: Config = {
  content: [
    // ...
    './node_modules/@crossed/ui/lib/**/*.{js,jsx,ts,tsx,md,mdx}',
    // ...
  ],
  // ...
};
export default config;/@crossed/ui/lib/**/*.{js,jsx,ts,tsx,md,mdx}',

```
### Postcss
Add code in your `postcss.config.js`

```javascript
module.exports = {
  plugins: {
    // ...
    'modify-selectors': {
      enable: true,
      modify: [
        {
          match: '*',
          with: (selector) => {
            const regEx = new RegExp('^\\.', 'g');
            if (selector.match(regEx)) {
              return `${selector}, [data-class-name~="${selector.replace(
                regEx,
                ''
              )}"]`;
            }
            return selector;
          },
        },
      ],
    },
    // ...
  },
}

```

### Add CrossedProvider
```typescript
import { CrossedProvider } from "@crossed/ui"

export default function App({ Component, pageProps }) {
  return (
    <CrossedProvider defaultTheme={'dark'}>
      <Component {...pageProps} />
    </CrossedProvider>
  );
}
```
