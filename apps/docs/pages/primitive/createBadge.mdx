    ## Usage
    ```tsx
      import type { ComponentProps } from "react";
      import { createBadge } from "@crossed/primitive";

      const Badge = createBadge({
        Root: (props: ComponentProps<"div">) => <div {...props} />,
        Text: (props: ComponentProps<"p">) => <p {...props} />,
      })

      const { Text } = Badge;

      function App() {
        return (
          <Badge>
            <Text>
              Hello
            </Text>
          </Badge>
        );
      }
    ```
