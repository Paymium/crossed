import { RovingFocus } from '@crossed/primitive';
import { useState } from 'react';

export const Tata = () => {
  const [random, setRandom] = useState(false);
  return (
    <>
      <button onClick={() => setRandom((e) => !e)}>show</button>
      <RovingFocus>
        <div>
          {[...Array(3).keys()].map((e) => (
            <RovingFocus.Item key={e}>
              <button className="w-5 h-5 bg-neutral-500 mx-1" />
            </RovingFocus.Item>
          ))}
          {random && (
            <RovingFocus.Item>
              <button className="w-5 h-5 bg-neutral-500 mx-1" />
            </RovingFocus.Item>
          )}
          {[...Array(3).keys()].map((e) => (
            <RovingFocus.Item key={e}>
              <button className="w-5 h-5 bg-neutral-500 mx-1" />
            </RovingFocus.Item>
          ))}
        </div>
      </RovingFocus>
    </>
  );
};
