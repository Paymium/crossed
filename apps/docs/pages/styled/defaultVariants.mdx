import { ButtonDefaultVariantsDemo, ButtonDefaultVariantsNativeDemo } from '@crossed/demo';
import { CodeDemo, Code } from 'components/CodeDemo';
import { getStaticDemo } from 'components/getStaticDemo';
import { Tabs as TabsNext, Tab as TabNext } from 'nextra-theme-docs';

export const getStaticProps = async (e) => {
  return {
    props: {
      ssg: {
        defaultVariants: (
          await getStaticDemo('styled/button/DefaultVariants')(e)
        ).props.ssg.code,
        defaultVariantsNative: (
          await getStaticDemo('styled/button/DefaultVariants.native')(e)
        ).props.ssg.code,
      },
    },
  };
};
