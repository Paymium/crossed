import { ButtonVariantsDemo, ButtonVariantsNativeDemo } from '@crossed/demo';
import { CodeDemo, Code } from 'components/CodeDemo';
import { getStaticDemo } from 'components/getStaticDemo';
import { Tabs as TabsNext, Tab as TabNext } from 'nextra-theme-docs';

export const getStaticProps = async (e) => {
  return {
    props: {
      ssg: {
        variants: (await getStaticDemo('styled/button/Variants')(e)).props.ssg
          .code,
        variantsNative: (
          await getStaticDemo('styled/button/Variants.native')(e)
        ).props.ssg.code,
      },
    },
  };
};

# Variants

You can create variants with `variants` param, It use base `className`, `props` and `state`.  
See [Base](/styled/base) for more explanation
