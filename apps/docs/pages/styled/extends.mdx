import { ButtonExtendsDemo, ButtonExtendsNativeDemo } from '@crossed/demo';
import { CodeDemo, Code } from 'components/CodeDemo';
import { getStaticDemo } from 'components/getStaticDemo';
import { Tabs as TabsNext, Tab as TabNext } from 'nextra-theme-docs';

export const getStaticProps = async (e) => {
  return {
    props: {
      ssg: {
        extends: (await getStaticDemo('styled/button/Extends')(e)).props.ssg
          .code,
        extendsNative: (await getStaticDemo('styled/button/Extends.native')(e)).props.ssg
          .code,
      },
    },
  };
};

# Extends component

All styled components have function `.styles(variantsParams)`  
If we take our example, `Button.styles` return theme with state correspond to params sent.
